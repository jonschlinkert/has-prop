## Install
Install with [npm](https://www.npmjs.com/):

```sh
$ npm i has-prop --save
```


## Usage

```js
var hasProp = require('has-prop');

hasProp({foo: null}, 'foo');
//=> true

hasProp({foo: {bar: ''}}, 'foo.bar');
//=> true

hasProp({foo: null}, 'bar');
//=> false

hasProp({foo: {bar: ''}}, 'foo.baz');
//=> false
```
